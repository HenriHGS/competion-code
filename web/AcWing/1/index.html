    class Solution {
        public:
            vector twoSum(vector& nums, int target) {
                vector ans;
                map ma;
                int pos = 0;
                for(auto x: nums){
                    int y = target - x;
                    if(ma.count(y)) {
                        ans.push_back(ma[y]);
                        ans.push_back(pos);
                        break;
                    }
                    ma[x] = pos ++ ;
                }
                return ans;
            }
        };
